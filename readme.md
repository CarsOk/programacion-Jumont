## Ejercicio de excel
```
Sub ejemplo()

    a = InputBox("colocar primer numero")
    b = InputBox("colocar segundo numero")
    c = InputBox("colocar tercer numero")
    d = InputBox("colocar cuarto numero")
    e = Int(a) + Int(b) + Int(c) + Int(d)
    MsgBox "El promedio es" & e / 4



End Sub
```
